def get_pcroom_table():
    driver = webdriver.Chrome('./chromedriver')
    driver.implicitly_wait(30)
    url = "https://benq4ever.blog.me/221196878240"
    driver.get(url)
    driver.switch_to.frame(0)
    driver.switch_to.frame("mainFrame")
    html = driver.page_source
    soup = BeautifulSoup(html, 'html.parser')
    pcroom_table = soup.select_one('div#postViewArea div.view table:nth-of-type(2)')
    return pcroom_table

앞선 포스팅에서 네이버 블로그에 있는 벤큐 피시방 목록을 크롤링 하는데에는 성공했다. 이제 크롤링한 정보를 db에 pcroom모델 객체로 저장해야 된다. pcroom의 모델은 다음과 같다.

class PCroom(models.Model):
    createtime_datetime = models.DateTimeField(auto_now_add=True, blank=True)
    name_char = models.CharField(max_length=20)
    monitor_char = models.CharField(max_length=20)
    address_char = models.CharField(max_length=40)
    url_char = models.CharField(max_length=30, blank=True, null=True)
    longitude_float = models.FloatField(max_length= 10, null=True)
    latitude_float = models.FloatField(max_length=10, null=True)

대강 이런 에러가 뜬다. script.py에 장고 환경 변수가 설정이 안되있다는 뜻이다. 이를 해결하기 위해 django standalone이 있다. django standalone은 일반 파이썬 스크립트에서도 ORM 같은 장고 기능을 사용할 수 있게해준다. django standalone을 사용하기 위해서는 파일 상단에 다음 4줄만 추가해 주면 된다.

# script.py
import os
import django
os.environ.setdefault('DJANGO_SETTINGS_MODULE', 'config.settings') # 프로젝트 폴더에 있는 wsgi.py에서 가져온다.
django.setup() # 장고 셋팅값을 불러온다.
from myapp.models import PCroom
pcroom = PCroom()
pcroom.name_char = "7 피시방"
pcroom.moniter_char = "benq 24인치"
pcroom.address_char = "서울시 광진구 아차산로"
# ...
pcroom.save() # 성공!

def to_location(address): # 네이버 api를 이용해 도로명 주소를 위도 경도로 변환
    url = "https://naveropenapi.apigw.ntruss.com/map-geocode/v2/geocode?query="+address+"&coordinate=11,10"
    custom_headers = {
        "X-NCP-APIGW-API-KEY-ID" : 'YOUR_KEY_ID',
        "X-NCP-APIGW-API-KEY" : "YOUR_KEY"
    }
    req=requests.get(url, headers=custom_headers)
    try:
        return (req.json()["addresses"][0]["x"], req.json()["addresses"][0]["y"])
    except:
        print(address, "좌표화 불가.")

def save_pcroom_table():
    pcroom_table = get_pcroom_table() # PC방 목록 불러옴
    rows = pcroom_table.select('tr')
    for row in rows[1:]:
        pcroom = PCroom()
        cells = row.select('td')
        pcroom.name_char = cells[1].text
        pcroom.monitor_char = cells[2].text
        pcroom.address_char = cells[3].text
        pcroom.url_char = cells[4].text
        location_point = to_location(pcroom.address_char)
        if location_point:
            pcroom.longitude_float = location_point[0]
            pcroom.latitude_float = location_point[1]
        pcroom.save() # PC방 객체 저장

django standalone을 몰랐을 때는 sqlite db에 직접 접근해서 쿼리문으로 PCroom객체들을 저장했다. 잘 작동했지만 문제가 있었다. 배포할 때는 postgres같은 다른 db를 사용해야 하는데 sqlite 쿼리문으로는 데이터를 저장할 수 없는 것이다. django standalone은 외부 파일에서 ORM말고 다른 장고 기능을 사용할 때에도 요긴하게 쓸 수 있을것 같다.
